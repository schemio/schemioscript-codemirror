# Simple expression

a1 +3

==>

Program(BinaryExpression(VariableName,ArithOp,Number))


# Simple expression2

(a1+(2-3))

==>

Program(
    ParenthesizedExpression(
        BinaryExpression(
            VariableName,ArithOp,ParenthesizedExpression(
                BinaryExpression(Number,ArithOp,Number)
            )
        )
    )
)


# Simple expression with whitespace

(a + b * 3 - w^2%3)

==>

Program(
    ParenthesizedExpression(
        BinaryExpression(
            BinaryExpression(
                VariableName,ArithOp,BinaryExpression(
                    VariableName,ArithOp,Number
                )
            ),
            ArithOp,BinaryExpression(
                VariableName,ArithOp,BinaryExpression(
                    Number,ArithOp,Number
                )
            )
        )
    )
)


# Assigment expression

qwe = x* 3 +4

==>

Program(
    AssignmentExpression(
        VariableName,
        BinaryExpression(
            BinaryExpression(VariableName,ArithOp,Number),
            ArithOp,
            Number
        )
    )
)
